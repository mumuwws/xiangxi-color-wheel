            onSpinEnd();

            return;

        }

        animFrame = requestAnimationFrame(animateSpin);

    }

    

    function startSpin() {

        if (spinning) return;

        spinVelocity = Math.random() * 0.5 + 0.3;

        spinning = true;

        animateSpin();

    }

    

    function drawWheel() {

        if (!ctx) return;

        ctx.clearRect(0, 0, width, height);

        const angleStep = (Math.PI * 2) / segmentLabels.length;

        for (let i = 0; i < segmentLabels.length; i++) {

            const start = i * angleStep + currentAngle;

            const end = start + angleStep;

            ctx.beginPath();

            ctx.moveTo(centerX, centerY);

            ctx.arc(centerX, centerY, radius, start, end);

            ctx.fillStyle = segmentColors[i % segmentColors.length];

            ctx.fill();

            ctx.strokeStyle = "white";

            ctx.lineWidth = 2;

            ctx.stroke();

            ctx.fillStyle = "#2c3e2f";

            ctx.font = `bold ${Math.floor(radius * 0.08)}px "Segoe UI"`;

            const mid = start + angleStep / 2;

            const textRadius = radius * 0.7;

            const x = centerX + Math.cos(mid) * textRadius;

            const y = centerY + Math.sin(mid) * textRadius + 8;

            let shortText = segmentLabels[i].length > 4 ? segmentLabels[i].slice(0, 4) : segmentLabels[i];

            ctx.fillStyle = "#2d2b26";

            ctx.fillText(shortText, x - 12, y);

        }

        ctx.beginPath();

        ctx.moveTo(centerX - 12, 10);

        ctx.lineTo(centerX, 0);

        ctx.lineTo(centerX + 12, 10);

        ctx.fillStyle = "#e67e22";

        ctx.fill();

        ctx.beginPath();

        ctx.arc(centerX, centerY, radius * 0.06, 0, 2 * Math.PI);

        ctx.fillStyle = "#f5bc70";

        ctx.fill();

    }

    

    function handleCanvasClick(e) {

        if (spinning) return;

        const rect = canvas.getBoundingClientRect();

        const scaleX = canvas.width / rect.width;

        const scaleY = canvas.height / rect.height;

        const mouseX = (e.clientX - rect.left) * scaleX;

        const mouseY = (e.clientY - rect.top) * scaleY;

        const dx = mouseX - centerX;

        const dy = mouseY - centerY;

        const distance = Math.hypot(dx, dy);

        if (distance > radius) return;

        let clickAngle = Math.atan2(dy, dx);

        if (clickAngle < 0) clickAngle += Math.PI * 2;

        let trueAngle = (clickAngle - currentAngle + Math.PI * 2) % (Math.PI * 2);

        const step = (Math.PI * 2) / segmentLabels.length;

        let idx = Math.floor(trueAngle / step) % segmentLabels.length;

        const selectedColor = colorMap[idx];

        switchToColor(selectedColor.key);

        drawWheel();

    }

    

    function resizeCanvas() {

        const container = canvas.parentElement;

        const size = container.clientWidth;

        canvas.width = size;

        canvas.height = size;

        width = size;

        height = size;

        centerX = width / 2;

        centerY = height / 2;

        radius = width / 2 - 8;

        drawWheel();

    }
